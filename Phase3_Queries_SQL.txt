-- Indexes
-- Improves query performance on frequently searched columns
CREATE INDEX idx_child_lastname ON Child(LastName);
CREATE INDEX idx_donation_amount ON Donation(Amount);

-- Views
-- Provides simplified virtual tables for easier data access
CREATE VIEW Child_Event_View AS SELECT c.FirstName, c.LastName, e.EventName FROM Child c JOIN Child_Events ce ON c.ChildId = ce.ChildId JOIN Event e ON ce.EventId = e.EventId;
CREATE VIEW Donation_Summary AS SELECT DonorId, SUM(Amount) AS TotalDonation FROM Donation GROUP BY DonorId;

-- Business Requirement Queries
-- Answers practical questions based on system usage
-- Children with Foster parents
SELECT c.FirstName, fp.FamilyName FROM Child c JOIN Fostering f ON c.ChildId = f.ChildId JOIN Foster_Parent fp ON f.FosterParentId = fp.FosterParentId;
-- Total Donations Per Donor:
SELECT d.DonorName, SUM(dn.Amount) AS TotalDonated FROM Donor d JOIN Donation dn ON d.DonorId = dn.DonorId GROUP BY d.DonorName;
-- Events With Number of Volunteers
SELECT e.EventName, COUNT(ve.VolunteerId) AS TotalVolunteers FROM Event e JOIN Volunteer_Event ve ON e.EventId = ve.EventId GROUP BY e.EventName;
-- Children Receiving Help From Volunteers:
SELECT c.FirstName, v.FirstName AS VolunteerName FROM Child c JOIN Volunteer_Child vc ON c.ChildId = vc.ChildId JOIN Volunteer v ON vc.VolunteerId = v.VolunteerId;
-- Financial Records Summary
SELECT RecordType, COUNT(*) AS TotalRecords FROM Financial_Record GROUP BY RecordType;

-- Basic SELECT
-- Retrieves simple data from a table
SELECT FirstName, LastName, Status From Child;

-- Row Limiting
-- Restricts the number of rows returned
SELECT FirstName, LastName From Child Fetch FIRST 5 Rows ONLY;
SELECT * FROM Donation FETCH FIRST 3 ROWS ONLY;

-- Sorting
-- Orders results based on specific columns
SELECT FirstName, LastName,DateOfBirth FROM Child ORDER BY DateOfBirth DESC;
SELECT DonorName, DonorType FROM Donor ORDER BY DonorName ASC;

-- Filtering
-- Uses conditions to filter specific records
SELECT * FROM child WHERE (FirstName LIKE 'j%' AND Status = 'Active') OR Gender = 'F';
SELECT * FROM Volunteer WHERE (Skill LIKE '%Teach%' AND ContactInfo LIKE '%gmail%') OR FirstName LIKE 'A%';

-- Character Functions
-- Manipulates text data (e.g., upper/lower case)
SELECT UPPER(FirstName) AS UPPERName, LOWER(LastName) AS LowerName FROM Child;
SELECT FirstName, CONCAT(FirstName, ' ', LastName) AS FullName FROM Child;

-- Numeric Functions
-- Formats numeric values
SELECT DonationId, ROUND(Amount, 0) AS RoundedAmount FROM Donation;
SELECT DonationId, TRUNC(Amount, 0) AS TruncatedAmount FROM Donation;

-- Date Functions
-- Performs calculations on date values
SELECT FirstName, DateOfBirth, MONTHS_BETWEEN(SYSDATE, DateOfBirth)/12 AS Age FROM Child;
SELECT FirstName, ADD_MONTHS(DateOfBirth, 12) AS NextYearDate FROM Child;

-- Aggregate Functions
-- Performs calculations on groups of data
SELECT COUNT(*) AS TotalChildren, AVG(MONTHS_BETWEEN(SYSDATE, DateOfBirth)/12) AS AvgAge FROM Child;

-- GROUP BY and HAVING
-- Groups data and filters grouped results
SELECT Gender, Status, COUNT(*)  FROM Child GROUP BY Gender, Status;
SELECT DonorId, SUM(Amount) AS TotalDonated FROM Donation GROUP BY DonorId HAVING SUM(Amount) > 1000;
SELECT Gender, COUNT(*) AS Total FROM Child GROUP BY Gender HAVING COUNT(*) > 1;

-- Joins
-- Combines data from multiple tables
SELECT c.FirstName, c.LastName, e.EventName FROM Child c JOIN Child_Events ce ON c.ChildId = ce.ChildId JOIN Event e ON ce.EventId = e.EventId;
-- Left Join:
SELECT c.FirstName, e.EventName FROM Child c LEFT JOIN Child_Events ce ON c.ChildId = ce.ChildId LEFT JOIN Event e ON ce.EventId = e.EventId;
-- Right Join:
SELECT e.EventName, ce.ChildId FROM Event e RIGHT JOIN Child_Events ce ON e.EventId = ce.EventId;
-- Multi Join:
SELECT v.FirstName, v.LastName, c.FirstName AS ChildName FROM Volunteer v JOIN Volunteer_Child vc ON v.VolunteerId = vc.VolunteerId JOIN Child c ON vc.ChildId = c.ChildId;
-- Multi Join With Filter:
SELECT c.FirstName, e.EventName FROM Child c JOIN Child_Events ce ON c.ChildId = ce.ChildId JOIN Event e ON ce.EventId = e.EventId WHERE e.EventDate > SYSDATE;
-- Join With Condition:
SELECT v.FirstName, v.Skill, c.FirstName AS ChildName FROM Volunteer v JOIN Volunteer_Child vc ON v.VolunteerId = vc.VolunteerId 
JOIN Child c ON vc.ChildId = c.ChildId WHERE v.Skill = 'Teaching';

-- Subqueries
-- Queries nested inside other queries
-- Normal Subquery
SELECT DonorName FROM Donor WHERE DonorId IN ( SELECT DonorId FROM Donation WHERE Amount > 1000);
-- Scalar Subquery
SELECT FirstName,
       (
	 SELECT COUNT(*) 
         FROM Child_Events ce 
         WHERE ce.ChildId = c.ChildId
	) AS EventCount
FROM Child c;
-- NOT IN Subquery
SELECT FirstName, LastName FROM Child
WHERE ChildId NOT IN (
    SELECT ChildId FROM Child_Events
);
-- MAX Value Subquery:
SELECT DonorName FROM Donor WHERE DonorId IN (
    SELECT DonorId
    FROM Donation
    WHERE Amount = (SELECT MAX(Amount) FROM Donation)
);
-- Correlated Subquery
SELECT d.DonorName, d.DonorId FROM Donor d
WHERE EXISTS (
    SELECT 1
    FROM Donation dn
    WHERE dn.DonorId = d.DonorId
);
-- Exists Subquery
SELECT FirstName, LastName FROM Child c WHERE EXISTS ( SELECT 1 FROM Child_Events ce WHERE ce.ChildId = c.ChildId);

-- Extra Functionality
-- Includes advanced features such as procedures
-- Financial Query:
SELECT RecordType, SUM( 
    CASE WHEN DonationId IS NOT NULL THEN (SELECT Amount FROM Donation d WHERE d.DonationId = fr.DonationId)
    	WHEN SponsorshipId IS NOT NULL THEN (SELECT Amount FROM Sponsorship s WHERE s.SponsorshipId = fr.SponsorshipId)
    	ELSE 0
    END
) AS TotalAmount
FROM Financial_Record fr GROUP BY RecordType;

-- Join and Aggregate
SELECT e.EventName, COUNT(ce.ChildId) AS TotalParticipants FROM Event e JOIN Child_Events ce ON e.EventId = ce.EventId GROUP BY e.EventName;

-- Between
SELECT * FROM Donation WHERE Amount BETWEEN 500 AND 5000;

-- Distinct
SELECT DISTINCT Gender FROM Child;

-- NULL Check
SELECT * FROM Financial_Record WHERE DonationId IS NULL;

-- Order and Function
SELECT FirstName, LENGTH(FirstName) AS NameLength FROM Child ORDER BY NameLength DESC;

-- Complex Queries
SELECT c.FirstName, c.LastName, COUNT(ce.EventId) AS EventsAttended FROM Child c LEFT JOIN Child_Events ce ON c.ChildId = ce.ChildId GROUP BY c.FirstName, c.LastName
ORDER BY EventsAttended DESC;

SELECT c.FirstName, c.LastName, COUNT(vc.VolunteerId) AS VolunteerCount FROM Child c LEFT JOIN Volunteer_Child vc ON c.ChildId = vc.ChildId 
GROUP BY c.FirstName, c.LastName ORDER BY VolunteerCount DESC;

-- String Function
SELECT FirstName, SUBSTR(FirstName, 1, 3) AS ShortName FROM Child;

-- Simple Procedure
CREATE OR REPLACE PROCEDURE GetChildCount AS
    total NUMBER;
BEGIN
    SELECT COUNT(*) INTO total FROM Child;
    DBMS_OUTPUT.PUT_LINE('Total Children: ' || total);
END;
